### 2.2 DAO

```ts
// src/storage/topics.ts

export interface Topic {
  name: string
  keywords: string[]   // 解析后的数组（DAO 内部 parse JSON）
  description: string | null
  createdAt: string
}

export interface TopicStat {
  topic: string
  discussionCount: number
  firstDiscussedAt: string | null
  lastDiscussedAt: string | null
}

export function createTopicsDao(handle: DbHandle): TopicsDao
export function createTopicStatsDao(handle: DbHandle): TopicStatsDao

interface TopicsDao {
  list(): Topic[]
  get(name: string): Topic | null
  // add/remove 留给 v0.7 工具调用
}

interface TopicStatsDao {
  all(): TopicStat[]                                   // ORDER BY last_discussed_at DESC NULLS LAST
  get(topic: string): TopicStat | null
  incrementAndUpdate(topic: string, now: Date): void  // UPSERT: count++, first/last
}
```

### 2.3 UPSERT SQL

```sql
INSERT INTO topic_stats (topic, discussion_count, first_discussed_at, last_discussed_at)
VALUES (?, 1, ?, ?)
ON CONFLICT(topic) DO UPDATE SET
  discussion_count = discussion_count + 1,
  last_discussed_at = excluded.last_discussed_at;
```

`first_discussed_at` 第一次 INSERT 写入，UPDATE 不覆盖——符合"首次讨论时间"语义。

## 3. 算法

### 3.1 Jaccard 相似度

```ts
function jaccard(a: string[], b: string[]): number {
  if (a.length === 0 || b.length === 0) return 0
  const setA = new Set(a.map(s => s.toLowerCase()))
  const setB = new Set(b.map(s => s.toLowerCase()))
  let intersection = 0
  for (const x of setA) if (setB.has(x)) intersection++
  const union = setA.size + setB.size - intersection
  return union === 0 ? 0 : intersection / union
}
```

**大小写不敏感**：13 岁口语 keyword "Minecraft" / "minecraft" 都认。

### 3.2 matchTopic

```ts
export interface TopicMatch {
  topic: string          // topics.name
  jaccard: number        // matched score
  shared: string[]       // for debug
}

export function matchTopic(
  sessionKeywords: string[],
  topics: Topic[],
  threshold: number = 0.3,
): TopicMatch | null {
  if (sessionKeywords.length === 0) return null
  const normSession = sessionKeywords.map(s => s.toLowerCase())
  let best: TopicMatch | null = null
  for (const t of topics) {
    const shared = t.keywords.filter(k => normSession.includes(k.toLowerCase()))
    if (shared.length === 0) continue
    const score = jaccard(normSession, t.keywords)
    if (score < threshold) continue
    if (best === null || score > best.jaccard ||
        (score === best.jaccard && t.name < best.topic)) {
      best = { topic: t.name, jaccard: score, shared }
    }
  }
  return best
}
```

**tie-break**：score 相等时按 `topic.name` 字典序，result 确定性。L1 测试：构造两个完全相同的 topic，断言返回字典序小的。

### 3.3 注入 [System Context]

```ts
export function buildSystemContext(
  state: SessionState,
  lastReview: LastReview | null = null,
  activeTopics: TopicStat[] = [],
): string {
  const lines = ['[System Context]', ...]
  if (lastReview) {
    lines.push(...)
    lines.push(...)
  }
  if (activeTopics.length > 0) {
    const top5 = activeTopics.slice(0, 5)
    const parts = top5.map(t => {
      const daysAgo = computeDaysAgo(t.lastDiscussedAt)
      return `${t.topic} (${t.discussionCount} time${t.discussionCount === 1 ? '' : 's'}, ${formatDaysAgo(daysAgo)})`
    })
    lines.push(`- Active topics: ${parts.join(', ')}`)
  }
  return lines.join('\n')
}
```

`formatDaysAgo(n)`: `n === 0 → "today"`, `n === 1 → "yesterday"`, `n >= 2 → "N days ago"`

## 4. CLI 集成

### 4.1 启动

```ts
// 现有 lastReview 加载旁边
const lastReview: LastReview | null = loadLastReview(db)
const activeTopics: TopicStat[] = createTopicStatsDao(db).all()  // 启动时一次性读
// activeTopics 在整个 session 内不变（topic_stats 在 END 才更新）
```

### 4.2 主循环

```ts
const system = buildFinalSystem(systemPrompt, state, isFirstTurn ? lastReview : null, activeTopics)
```

**activeTopics 不分 first-turn**（与 lastReview 不同）——topic 视角是"这个学生最近聊过什么"，跨整轮都相关；lastReview 是"上次 session 具体回顾"，只在 first turn 出现防 stale。

### 4.3 退出（finally 块）

```ts
const allMessages = messages.getBySession(session.id)
let summaryText, summaryKeywords
try {
  const review = await summarize(allMessages.map(...), client)
  summaryText = review.summary
  summaryKeywords = review.keywords
} catch (err) {
  summaryText = '(summarization failed)'
  summaryKeywords = []
  process.stderr.write(`[cli] summarize failed: ...`)
}

// NEW v0.6
const topics = createTopicsDao(db).list()
const match = matchTopic(summaryKeywords, topics, 0.3)
if (match) {
  createTopicStatsDao(db).incrementAndUpdate(match.topic, new Date())
  process.stderr.write(`[cli] topic match: ${match.topic} jaccard=${match.jaccard.toFixed(2)} shared=[${match.shared.join(',')}]\n`)
} else {
  process.stderr.write(`[cli] topic match: none (keywords=[${summaryKeywords.join(',')}])`)
}

sessions.markEnded(session.id, { phaseHistory, summary, keywords, reason })
```

**fallback 行为**：如果 summarize 失败 → `summaryKeywords = []` → matchTopic 一定返 null → 不更新 topic_stats。不阻塞 markEnded。

## 9. 风险

| 风险 | 触发条件 | 处置 |
|---|---|---|
| **Jaccard 阈值 0.3 太松** | 学生聊 "I think also really" 噪音关键词匹到某 topic | L1 测 1 个反例 fixture；live demo 验证；v0.6.1 调阈值 |
| **Jaccard 阈值 0.3 太严** | 真聊 minecraft 但关键词分散 | seed 关键词 10 个覆盖常用词；后续看真实数据调 |
| **Migration 003 seed 跟未来工具添加的 topic 冲突** | v0.7 addTopic 时 PRIMARY KEY 冲突 | DAO 显式 catch + warn（v0.7 处理） |
| **activeTopics 在 session 中途不更新** | LLM 不知道"这次匹配上 minecraft 了" | 这是设计选择：activeTopics 是"session 外视角"，session 内行为不据此改变；v0.7 工具调用时再说 |
| **`loadTopicStats` 启动时读一次，长 session 内 stale** | session 30 min 内 topic_stats 不会被同进程改 | OK，v0.6 单用户单 session 不会同进程写 topic_stats |
| **keywords 数组含非英文 / 拼写错误** | "mc" / "creepr" 匹不上 minecraft | 接受现状；v0.7 LLM 工具可规范化（stemming / fuzzy） |
| **biome 模板字面量风格** | strings 拼接触发 lint | 沿用 v0.5 经验，最后 `biome check --write --unsafe` |
