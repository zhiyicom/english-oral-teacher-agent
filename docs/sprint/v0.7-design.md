### 2.2 DAO

```ts
// src/storage/mistakes.ts
import type { DbHandle } from './db.js'

export type MistakeCategory = 'grammar' | 'vocabulary' | 'spelling'

export interface Mistake {
  id: number
  sessionId: string
  original: string
  corrected: string
  category: MistakeCategory
  ts: string
}

export interface AppendMistakeInput {
  sessionId: string
  original: string
  corrected: string
  category: MistakeCategory
}

export interface MistakesDao {
  append(input: AppendMistakeInput): Mistake
  getBySession(sessionId: string): Mistake[]
  getRecent(limit: number, now?: Date): Mistake[]  // 跨 session，按 ts DESC
}

interface MistakeRow {
  id: number
  session_id: string
  original: string
  corrected: string
  category: string
  ts: string
}

function rowToMistake(row: MistakeRow): Mistake {
  return {
    id: row.id,
    sessionId: row.session_id,
    original: row.original,
    corrected: row.corrected,
    category: row.category as MistakeCategory,
    ts: row.ts,
  }
}

export function createMistakesDao(handle: DbHandle): MistakesDao {
  const { raw } = handle
  const insert = raw.prepare(`
    INSERT INTO mistakes (session_id, original, corrected, category)
    VALUES (?, ?, ?, ?)
  `)
  const selectBySession = raw.prepare(`
    SELECT id, session_id, original, corrected, category, ts
    FROM mistakes
    WHERE session_id = ?
    ORDER BY ts ASC, id ASC
  `)
  const selectRecent = raw.prepare(`
    SELECT id, session_id, original, corrected, category, ts
    FROM mistakes
    ORDER BY ts DESC, id DESC
    LIMIT ?
  `)
  return {
    append(input) {
      const info = insert.run(input.sessionId, input.original, input.corrected, input.category)
      return {
        id: info.lastInsertRowid as number,
        sessionId: input.sessionId,
        original: input.original,
        corrected: input.corrected,
        category: input.category,
        ts: new Date().toISOString(),  // 近似 SQL CURRENT_TIMESTAMP
      }
    },
    getBySession(sessionId) {
      return (selectBySession.all(sessionId) as MistakeRow[]).map(rowToMistake)
    },
    getRecent(limit) {
      return (selectRecent.all(limit) as MistakeRow[]).map(rowToMistake)
    },
  }
}
```

`getRecent` 留 v0.7.1 用（correction 注入）。

## 3. Tool 系统

### 3.1 Tool interface

```ts
// src/agent/tool-registry.ts
import { z } from 'zod'

export interface Tool {
  readonly name: string
  readonly description: string
  readonly schema: z.ZodSchema
  execute(args: unknown): unknown
}

export class ToolRegistry {
  private tools = new Map<string, Tool>()

  register(tool: Tool): void {
    if (this.tools.has(tool.name)) {
      throw new Error(`Tool "${tool.name}" already registered`)
    }
    this.tools.set(tool.name, tool)
  }

  get(name: string): Tool | undefined {
    return this.tools.get(name)
  }

  list(): Tool[] {
    return [...this.tools.values()]
  }
}
```

### 3.2 Parser

```ts
// src/agent/tool-parser.ts

export interface ParsedToolCall {
  name: string
  args: Record<string, unknown>
  rawMatch: string  // 整段 <tool>...</tool> 原文
}

const TOOL_REGEX = /<tool>(\w+)\((\{[\s\S]*?\})\)<\/tool>/

export function parseToolCall(response: string): ParsedToolCall | null {
  const match = TOOL_REGEX.exec(response)
  if (!match) return null
  const [, name, argsJson] = match
  let args: Record<string, unknown>
  try {
    args = JSON.parse(argsJson)
  } catch (e) {
    throw new Error(`Tool call args not valid JSON: ${(e as Error).message} (raw: ${argsJson})`)
  }
  return { name, args, rawMatch: match[0] }
}

export function stripToolCall(response: string, parsed: ParsedToolCall): string {
  return response.replace(parsed.rawMatch, '').trim()
}
```

`TOOL_REGEX` 设计：
- `<tool>` 字面
- `(\w+)` 抓 name
- `(` 字面
- `(\{[\s\S]*?\})` 非贪婪抓 args JSON（允许 nested braces）
- `)` 字面
- `</tool>` 字面

`stripToolCall` 只去掉"第一个匹配"——L1 测验证多次匹配时只 strip 第一个。

### 3.3 mark_mistake

```ts
// src/agent/tools/mark-mistake.ts
import { z } from 'zod'
import type { DbHandle } from '../../storage/db.js'
import { type MistakesDao, createMistakesDao } from '../../storage/mistakes.js'
import type { Tool } from '../tool-registry.js'

const MistakeArgsSchema = z.object({
  original: z.string().min(1).max(500),
  corrected: z.string().min(1).max(500),
  category: z.enum(['grammar', 'vocabulary', 'spelling']),
})

export function createMarkMistakeTool(db: DbHandle, sessionId: string): Tool {
  const mistakes: MistakesDao = createMistakesDao(db)
  return {
    name: 'mark_mistake',
    description:
      'Record a mistake the student just made. Use when the student says something ungrammatical, uses the wrong word, or misspells.',
    schema: MistakeArgsSchema,
    execute(args: unknown) {
      const parsed = MistakeArgsSchema.parse(args)
      return mistakes.append({ sessionId, ...parsed })
    },
  }
}
```

`sessionId` 在 CLI 启动时注入到 factory，LLM 不用关心。

## 4. System Prompt

### 4.1 新文件 `prompts/tools.md`

```markdown
# Tool Calling

You have one tool: `mark_mistake`. Call it when the student makes a grammar
mistake, uses the wrong word, or misspells. The tool records the mistake
to a database for future reference.

To call a tool, output EXACTLY this block on its own line, BEFORE your
spoken response to the student:

<tool>mark_mistake({"original": "<the student's wrong phrase>", "corrected": "<the corrected version>", "category": "grammar|vocabulary|spelling"})</tool>

After the tool block, you may continue speaking to the student (correcting
them gently, continuing the conversation, etc.). The tool block itself is
NOT shown to the student — it is a private signal to the system.

Only call this tool when you are CONFIDENT the student made a mistake
worth recording. Don't call it for stylistic preferences or things
that are debatable. One tool call per turn maximum.
```

### 4.3 prompt-builder 扩展

```ts
// src/agent/prompt-builder.ts
import { buildSystemString, loadToolsPrompt, type SystemPrompt } from '../prompts/loader.js'

export function buildFinalSystem(
  systemPrompt: SystemPrompt,
  state: SessionState,
  lastReview: LastReview | null = null,
  activeTopics: TopicStat[] = [],
  toolsPrompt: string = loadToolsPrompt(),  // 默认读盘，测试可注入
): string {
  return [
    buildSystemString(systemPrompt),
    '',
    buildSystemContext(state, lastReview, activeTopics),
    '',
    toolsPrompt,
  ].join('\n')
}
```

tools 段位置：在 [System Context] **之后**（end of system）。理由：tools 是"agent 能做什么"，放在最后做总结性补充；不放在 SOUL/AGENTS 之前是因为那是"agent 是谁"。

## 5. CLI 集成

### 5.1 启动

```ts
// cli.ts main() 启动段（v0.6 activeTopics 之后）
const mistakes = createMistakesDao(db)

toolRegistry.register(markMistake)
```

### 5.2 主循环（buffer + parse + execute + strip）

```ts
// v0.4-v0.6 现有 stream 写法：
// for await (const chunk of client.chatStream({...})) {
//   if (chunk.type === 'text') {
//     process.stdout.write(chunk.delta)  // ← 改
//     response += chunk.delta
//   }
// }
// process.stdout.write('\n\n')

// v0.7 改：
let response = ''
for await (const chunk of client.chatStream({...})) {
  if (chunk.type === 'text') {
    response += chunk.delta  // 只 buffer，不 stdout
  }
}

// 调完 LLM 后再处理 tool + 输出
let display = response
try {
  const parsed = parseToolCall(response)
  if (parsed) {
    display = stripToolCall(response, parsed)
    const tool = toolRegistry.get(parsed.name)
    if (tool) {
      try {
        tool.execute(parsed.args)
        process.stderr.write(
          `[cli] tool call: ${parsed.name}(${JSON.stringify(parsed.args)})\n`,
        )
      } catch (err) {
        process.stderr.write(
          `[cli] tool error in ${parsed.name}: ${(err as Error).message}\n`,
        )
      }
    } else {
      process.stderr.write(`[cli] tool unknown: ${parsed.name}\n`)
    }
  }
} catch (err) {
  process.stderr.write(`[cli] tool parse error: ${(err as Error).message}\n`)
}

process.stdout.write(`${display}\n\n`)
history.push({ role: 'assistant', content: response })  // history 存原始（含 tool 块）
messages.append({ sessionId: session.id, role: 'assistant', content: response })
```

**关键点**：
- `history` 存**原始** response（含 tool 块）——LLM 下轮能看到自己之前调用过什么
- `stdout` 给**剥掉** tool 块的 display ——学生看不到
- `messages.append` 也存原始——忠实记录 agent 的所有输出
- 三层 try/catch 隔离失败：parse error / unknown tool / execute error 都不会崩对话

### 5.3 finally 块

不动。v0.7 不影响 markEnded 流程。

### 6.1 mistake-yesterday.json shape

```json
{
  "name": "mistake-yesterday",
  "match": {"userSaysContains": "yesterday"},
  "chunks": [
    {"type": "text", "delta": "<tool>mark_mistake({\"original\": \"I go to school yesterday\", \"corrected\": \"I went to school yesterday\", \"category\": \"grammar\"})</tool>"},
    {"type": "text", "delta": "Oh I see, you went to school! What did you do there?"}
  ],
  "result": {
    "content": "<tool>mark_mistake({\"original\": \"I go to school yesterday\", \"corrected\": \"I went to school yesterday\", \"category\": \"grammar\"})</tool> Oh I see, you went to school! What did you do there?"
  }
}
```

**注意 fixture 匹配**：
- "yesterday" 也会出现在 stop 之前某些 fixture（"I want to stop yesterday"）—— 不会，`stop.json` 的 match 是 `"stop"`，first-match-wins 按文件名字母序：`creeper.json` 在前 → "yesterday" 不在 creeper 内容里 → `mistake-yesterday.json` 会命中
- 实际看现有 fixture 内容：`greeting.json`/`doing-fine.json`/`hobby-played.json`/`castle.json`/`creeper.json`/`stop.json`/`summarize.json` 都不含 "yesterday" 关键词
- 所以 `mistake-yesterday.json` 的 "yesterday" 匹配是唯一的

## 9. 风险与处置

| 风险 | 影响 | 处置 |
|---|---|---|
| LLM 不按格式输出 | real LLM 漏 `<tool>` | prompt 强约束；live demo 验 |
| LLM 输出多个 `<tool>` | 多 tool 解析 | 取第一个；prompt "One tool per turn" |
| stream → buffer 改 breaking | 现有 L3 测 | 现有测只看最终 stdout，buffer 模式 OK |
| Tool execute 抛错阻塞对话 | DB 故障 | 三层 try/catch + stderr 告警 + 跳过 |
| Fixture 误匹配 "yesterday" | Replay 顺序 | 检查现有 fixture 都不含 "yesterday"；新文件唯一 |
| sessionId 跨 session 污染 | tool 不纯 | factory 模式 + CLI 启动时绑定 |
| biome 模板字面量风格 | lint 失败 | `biome check --write` 自动修 |
| buffer 改 streaming → 真实 LLM 流式体验 | 用户看不到渐进输出 | LLM 响应本来就短（≤200 token），延迟 < 1s；可接受 v0.7 trade-off |
| Tool prompt 文件不存在 → loader 抛错 | CI 失败 | fixture 验证 prompts/tools.md 在仓 |
