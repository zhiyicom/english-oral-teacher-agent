### 3.1 A — Error classification + retry + fallback

`src/llm/errors.ts`：

```ts
export type LLMErrorClass =
  | 'rate_limit'      // 429
  | '5xx'             // 500-599
  | '4xx'             // 400-428, 430-599 (non-retryable client error)
  | 'network'         // fetch throws, no .status
  | 'content_filter'  // refusal in response (not in err; checked after stream)
  | 'unknown'

export interface ClassifiedError {
  classification: LLMErrorClass
  retryable: boolean
  original: Error
}

export function classifyLLMError(err: unknown): ClassifiedError {
  if (err && typeof err === 'object') {
    const e = err as { status?: number; code?: string; message?: string }
    // Anthropic SDK APIError has .status (HTTP code)
    if (typeof e.status === 'number') {
      if (e.status === 429) return { classification: 'rate_limit', retryable: true, original: err as Error }
      if (e.status >= 500) return { classification: '5xx', retryable: true, original: err as Error }
      if (e.status >= 400) return { classification: '4xx', retryable: false, original: err as Error }
    }
    // No status → network error (TypeError, ECONNRESET, etc.)
    return { classification: 'network', retryable: true, original: err as Error }
  }
  return { classification: 'unknown', retryable: false, original: new Error(String(err)) }
}
```

`src/cli.ts` (replaces direct `for await` over `chatStream`)：

```ts
interface StreamResult { response: string; usage: UsageChunk | null }

async function chatStreamWithRetry(
  client: LLMClient,
  opts: ChatOpts,
  maxAttempts = 2,  // 1 initial + 1 retry
): Promise<StreamResult> {
  let lastErr: unknown = null
  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    try {
      let response = ''
      let usage: UsageChunk | null = null
      for await (const chunk of client.chatStream(opts)) {
        if (chunk.type === 'text') response += chunk.delta
        else if (chunk.type === 'usage') usage = chunk
      }
      return { response, usage }
    } catch (err) {
      lastErr = err
      const classified = classifyLLMError(err)
      process.stderr.write(
        `[cli] llm error: ${classified.classification} (${(err as Error).message})\n`,
      )
      if (!classified.retryable) break
      if (attempt < maxAttempts) {
        process.stderr.write(`[cli] retrying in 1s... (attempt ${attempt + 1}/${maxAttempts})\n`)
        await new Promise((r) => setTimeout(r, 1000))
      }
    }
  }
  throw lastErr
}
```

### 3.2 B1 — Anchor pair

`src/agent/truncate-history.ts`：

```ts
export interface TruncateOptions {
  systemSize?: number
  /** First user/assistant pair; protected from being dropped. If absent, behaves as v0.7.5. */
  anchorPair?: readonly Message[]
}

export interface TruncateResult {
  messages: Message[]
  /** Number of (user, assistant) pairs dropped from the droppable portion. */
  dropped: number
}

export function truncateHistory(
  messages: Message[],
  budget: number,
  options: TruncateOptions = {},
): TruncateResult {
  const { systemSize = 0, anchorPair = [] } = options

  // Strip anchor pair from the front; the rest is droppable.
  let droppable = messages
  let protected: Message[] = []
  if (anchorPair.length > 0 && messages.length >= anchorPair.length) {
    const head = messages.slice(0, anchorPair.length)
    const headMatches = head.every(
      (m, i) => m.content === anchorPair[i].content && m.role === anchorPair[i].role,
    )
    if (headMatches) {
      protected = head
      droppable = messages.slice(anchorPair.length)
    }
  }

  let current = droppable
  let dropped = 0
  while (current.length > 2) {
    const total = estimateMessagesTokens([...protected, ...current]) + systemSize
    if (total <= budget) break
    current = current.slice(2)
    dropped += 1
  }
  return { messages: [...protected, ...current], dropped }
}
```

import type { Tool } from '../tool-registry.js'

export const SummarizeHistoryArgsSchema = z.object({
  target_tokens: z.number().int().min(100).max(3000).default(500),
})

export type SummarizeHistoryArgs = z.infer<typeof SummarizeHistoryArgsSchema>

const DESCRIPTION =
  'Compress the older part of the current conversation history to fit within ' +
  'target_tokens. Use when the conversation is getting long. The system will ' +
  'replace older messages with a summary and make a 2nd LLM call so you can ' +
  'respond. Do NOT call any tool in your 2nd response.'

export interface SummarizeHistoryResult {
  kind: 'summarize_history'
  targetTokens: number
}

/**
 * Marker tool. CLI does the actual history rewrite (history lives in cli.ts
 * scope, not in this tool). The tool just validates the args and signals
 * intent so CLI knows to invoke the summarizer and rewrite history before
 * the 2nd call. Matches the v0.7.3 A+B hybrid pattern.
 */
export function createSummarizeHistoryTool(): Tool {
  return {
    name: 'summarize_history',
    description: DESCRIPTION,
    schema: SummarizeHistoryArgsSchema,
    async execute(args: unknown): Promise<SummarizeHistoryResult> {
      const parsed = SummarizeHistoryArgsSchema.parse(args)
      return { kind: 'summarize_history', targetTokens: parsed.target_tokens }
    },
  }
}

### 3.4 B3 — Per-segment token cost logging

`src/agent/context-injector.ts`：

```ts
export interface SystemContextResult {
  text: string
  segments: {
    phase: number       // est tokens
    last: number        // 0 if no lastReview
    relevant: number    // 0 if no relevantPast
    active: number      // 0 if no activeTopics
    mistakes: number    // 0 if no recentMistakes
  }
}

export function buildSystemContext(
  state: SessionState,
  lastReview: LastReview | null = null,
  activeTopics: TopicStat[] = [],
  recentMistakes: Mistake[] = [],
  relevantPast: RelevantSession[] = [],
  now: Date = new Date(),
): SystemContextResult {
  // Build each segment separately to track sizes
  const lastTransitionAgo = Math.max(0, state.elapsedMin - state.lastTransitionAt)
  const phaseLines = [
    `[System Context]`,
    `- Phase: ${state.phase}`,
    `- Elapsed: ${state.elapsedMin.toFixed(1)} min`,
    `- Silence: ${state.silenceMin.toFixed(1)} min`,
    `- Last transition: ${lastTransitionAgo.toFixed(1)} min ago (entered ${state.phase})`,
  ]
  const phaseSeg = phaseLines.join('\n')
  const lines: string[] = [phaseSeg]

  let lastSeg = ''
  if (lastReview) {
    const dayWord = lastReview.daysAgo === 1 ? 'day' : 'days'
    const durStr = lastReview.durationMin != null ? `${lastReview.durationMin} min` : 'unknown'
    lastSeg = [
      `- Last session (${lastReview.daysAgo} ${dayWord} ago, ${durStr}): ${lastReview.summary}`,
      `- Last session keywords: ${lastReview.keywords.join(', ')}`,
    ].join('\n')
    lines.push(lastSeg)
  }

  let relevantSeg = ''
  if (relevantPast.length > 0) {
    const top = relevantPast.slice(0, 2)
    relevantSeg = [`- Relevant past sessions (N=${top.length}):`]
    for (const r of top) {
      const truncated = r.summary.length > 80 ? `${r.summary.slice(0, 80)}...` : r.summary
      const kwStr = r.keywords.join(', ')
      const dayWord = r.daysAgo === 1 ? 'day' : 'days'
      relevantSeg.push(`  - ${r.daysAgo} ${dayWord} ago: "${truncated}" (keywords: ${kwStr})`)
    }
    relevantSeg = relevantSeg.join('\n')
    lines.push(relevantSeg)
  }

  let activeSeg = ''
  if (activeTopics.length > 0) {
    const top = activeTopics.slice(0, 5)
    const parts = top.map((t) => {
      const daysAgo = computeDaysAgo(t.lastDiscussedAt, now)
      const timeWord = t.discussionCount === 1 ? 'time' : 'times'
      return `${t.topic} (${t.discussionCount} ${timeWord}, ${formatDaysAgo(daysAgo)})`
    })
    activeSeg = `- Active topics: ${parts.join(', ')}`
    lines.push(activeSeg)
  }

  let mistakesSeg = ''
  if (recentMistakes.length > 0) {
    const top = recentMistakes.slice(0, 5)
    mistakesSeg = [`- Recent mistakes (N=${top.length}):`]
    for (const m of top) {
      mistakesSeg.push(`  - "${m.original}" → "${m.corrected}" (${m.category})`)
    }
    mistakesSeg = mistakesSeg.join('\n')
    lines.push(mistakesSeg)
  }

  return {
    text: lines.join('\n'),
    segments: {
      phase: estimateTokens(phaseSeg),
      last: lastSeg ? estimateTokens(lastSeg) : 0,
      relevant: relevantSeg ? estimateTokens(relevantSeg) : 0,
      active: activeSeg ? estimateTokens(activeSeg) : 0,
      mistakes: mistakesSeg ? estimateTokens(mistakesSeg) : 0,
    },
  }
}
```

### 3.5 B4 — cache_control on messages

`src/llm/anthropic.ts` (在构造 Anthropic SDK message params 时)：

```ts
function toAnthropicMessages(messages: Message[]): Anthropic.MessageParam[] {
  // Apply cache_control to the last 3 messages (breaks cache into smaller chunks;
  // each subsequent turn hits cache on the prefix up to the breakpoint).
  // Wrap in try/catch: SDK may not support cache_control on message content blocks.
  try {
    return messages.map((m, i) => {
      const isBreakpoint = i >= messages.length - 3
      if (isBreakpoint) {
        return {
          role: m.role,
          content: [{
            type: 'text' as const,
            text: m.content,
            cache_control: { type: 'ephemeral' as const },
          }],
        }
      }
      return { role: m.role, content: m.content }
    })
  } catch (err) {
    // Fallback: plain messages (no cache_control on messages)
    process.stderr.write(`[llm] cache_control on messages unsupported: ${(err as Error).message}\n`)
    return messages.map((m) => ({ role: m.role, content: m.content }))
  }
}
```

> **Note**: This is best-effort. Live demo will verify whether SDK 0.41 + MiniMax-M3 actually support message-level cache_control. If not, the try/catch fallback keeps the system block cache_control working (already verified ✅).

### 3.6 D1-D4 — Topic engine

`src/agent/topic-engine.ts`：

```ts
import type { Topic, TopicStat } from '../storage/topics.js'

export interface TopicCandidate {
  topic: Topic
  /** Lower is better (less discussed). null = never discussed. */
  count: number | null
  /** Number of keywords matching student interests. */
  interest: number
  /** Final combined score. */
  score: number
}

/** D1 — exclude topics discussed in the last `excludeDays` days. */
export function filterHardExclude(
  topics: Topic[],
  stats: TopicStat[],
  excludeDays: number,
  now: Date,
): Topic[] {
  const cutoff = now.getTime() - excludeDays * 86_400_000
  return topics.filter((t) => {
    const stat = stats.find((s) => s.topic === t.name)
    if (!stat || !stat.lastDiscussedAt) return true  // never discussed → keep
    return Date.parse(stat.lastDiscussedAt) < cutoff
  })
}

/** D2 — sort by count ascending (less discussed first). Stable: alphabetical tiebreak. */
export function sortByCountAsc(topics: Topic[], stats: TopicStat[]): Topic[] {
  return [...topics].sort((a, b) => {
    const ca = stats.find((s) => s.topic === a.name)?.discussionCount ?? 0
    const cb = stats.find((s) => s.topic === b.name)?.discussionCount ?? 0
    if (ca !== cb) return ca - cb
    return a.name.localeCompare(b.name)
  })
}

/** D3 — compute interest overlap (case-insensitive). */
export function computeInterest(topic: Topic, interests: readonly string[]): number {
  if (interests.length === 0) return 0
  const norm = new Set(interests.map((s) => s.toLowerCase()))
  return topic.keywords.filter((k) => norm.has(k.toLowerCase())).length
}

/** Top-level entry — D1+D2+D3+D4 in one call. RNG injectable for testing. */
export function selectTopic(opts: {
  topics: Topic[]
  stats: TopicStat[]
  interests: readonly string[]
  excludeDays?: number
  now?: Date
  rng?: () => number
}): Topic | null {
  const excludeDays = opts.excludeDays ?? 30
  const now = opts.now ?? new Date()
  const rng = opts.rng ?? Math.random

  // D1: hard exclude
  let pool = filterHardExclude(opts.topics, opts.stats, excludeDays, now)
  if (pool.length === 0) return null

  // D2: sort by count ascending
  pool = sortByCountAsc(pool, opts.stats)

  // D3 + D4: score = -count * 0.1 + interest * 0.5 + noise(±0.2)
  const COUNT_WEIGHT = 0.1
  const INTEREST_WEIGHT = 0.5
  const NOISE_RANGE = 0.2

  const scored: TopicCandidate[] = pool.map((t) => {
    const stat = opts.stats.find((s) => s.topic === t.name) ?? null
    const count = stat?.discussionCount ?? 0
    const interest = computeInterest(t, opts.interests)
    const noise = (rng() * 2 - 1) * NOISE_RANGE
    return {
      topic: t,
      count,
      interest,
      score: -count * COUNT_WEIGHT + interest * INTEREST_WEIGHT + noise,
    }
  })

  scored.sort((a, b) => b.score - a.score)
  return scored[0].topic
}
```

### 3.7 D5 — topic_select tool

`src/agent/tools/topic-select.ts`：

```ts
import { z } from 'zod'
import type { Topic, TopicStat } from '../../storage/topics.js'
import { selectTopic } from '../topic-engine.js'
import type { Tool } from '../tool-registry.js'

export const TopicSelectArgsSchema = z.object({
  phase: z.enum(['WARM_UP', 'MAIN_ACTIVITY', 'WRAP_UP', 'END']).default('WARM_UP'),
  exclude_recent_days: z.number().int().min(0).max(365).default(30),
})

export type TopicSelectArgs = z.infer<typeof TopicSelectArgsSchema>

export interface TopicSelectResult {
  slug: string
  title: string
  est_minutes: number
}

const DESCRIPTION =
  'Pick a topic for the current conversation phase. Uses: ' +
  '(1) hard exclude topics discussed in the last N days, ' +
  '(2) soft preference for topics with low discussion count, ' +
  '(3) interest boost if student.interests matches, ' +
  '(4) weighted random selection to avoid deterministic picks. ' +
  'Returns the topic slug, title, and estimated minutes.'

export function createTopicSelectTool(opts: {
  topics: Topic[]
  stats: TopicStat[]
  interests: string[]
  rng?: () => number
}): Tool {
  return {
    name: 'topic_select',
    description: DESCRIPTION,
    schema: TopicSelectArgsSchema,
    execute(args: unknown): TopicSelectResult | { error: string } {
      const parsed = TopicSelectArgsSchema.parse(args)
      const winner = selectTopic({
        topics: opts.topics,
        stats: opts.stats,
        interests: opts.interests,
        excludeDays: parsed.exclude_recent_days,
        rng: opts.rng,
      })
      if (!winner) {
        return { error: 'No topics available after hard exclude' }
      }
      // est_minutes: hardcoded 15 (Topic schema doesn't have this field;
      // F7 topic library ≥20 will add it to Topic frontmatter — out of scope v0.7.6)
      return { slug: winner.name, title: winner.name, est_minutes: 15 }
    },
  }
}
```

**CLI 改动** (主循环加 `else if (parsed.name === 'topic_select')` 分支，结构跟 `memory_search` / `summarize_history` 一致；不做 history 改写，只走 2nd-call A+B 路径)。

## 6. 风险细化

| 风险 | 触发 | 缓解 |
|---|---|---|
| **A: 错误分类遗漏 SDK 边界情况** | Anthropic SDK throw 别的异常类型 | `classifyLLMError` default branch = 'unknown' retryable=false；L1 测试覆盖 5 类即可，不追求 100% |
| **A2: retry 把 transient 升级成永久** | 网络持续断 1 秒 + retry 也断 | retry 1 次上限；失败立即 fallback（A3）|
| **A3: auto-save 也失败** | DB 损坏 / 写盘 IO 错误 | fallback 的 try/catch 用 `(summarization failed)` placeholder + 仍写库（即使空 summary）|
| **B1: anchor pair 内容匹配实现脆弱** | if `messages[0..1].content !== anchorPair.content` → no protection | 用 content + role 比较；如果不匹配 log warn 一次 + 降级为 v0.7.5 行为 |
| **B1: anchor pair 本身就 > budget** | first pair 异常长（罕見）| 检测：如果 anchor + 2 个 oldest > budget → log warn + 不 truncate（保留全部历史）|
| **B2: history rewrite 丢 anchor pair** | 实现错误导致 firstPair 被覆盖 | CLI 用 `history.length = 0; push(...firstPair, ...)` 显式保留 |
| **B2: 2nd-call LLM 也调 tool** | LLM 不听话 | safety strip 已实现（`followupParsed` 检查）；L3 fixture 也只产生无 tool 的响应 |
| **B2: marker `[v076_history_summary]` 跟 `[tool_result_v073]` 冲突** | fixture 匹配冲突 | marker 字符串明显不同（v076 vs v073）；L3 测试验证 |
| **B3: context-injector 改返回类型破现有 callers** | prompt-builder.ts 还在用 `buildSystemContext().text` 风格 | prompt-builder 同时加 `buildFinalSystemSegments()` 包装函数，CLI 用新函数；旧 `buildFinalSystemSplit` 保留（用 `.text` 字段）|
| **B4: SDK 0.41 不支持 cache_control on messages** | 实测发现 SDK reject | try/catch fallback 到无 cache_control；log warn 一次；system block cache_control 仍工作（已验证）|
