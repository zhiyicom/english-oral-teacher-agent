### 3.3 `web/src/lib/api.ts` — typed fetch wrapper

```ts
import type { SessionApi } from './types'

// BASE is empty in dev (Vite proxy) and prod (same-origin after build).
// This means CORS never arises; all browser fetches go to its own origin.
const BASE = ''

export async function listSessions(): Promise<SessionApi[]> {
  const res = await fetch(`${BASE}/api/sessions`)
  if (!res.ok) throw new Error(`listSessions: HTTP ${res.status}`)
  const body = (await res.json()) as { sessions: SessionApi[] }
  return body.sessions
}

export async function createSession(): Promise<{ id: string }> {
  const res = await fetch(`${BASE}/api/sessions`, { method: 'POST' })
  if (!res.ok) throw new Error(`createSession: HTTP ${res.status}`)
  return (await res.json()) as { id: string }
}

export async function getSession(id: string): Promise<SessionApi> {
  const res = await fetch(`${BASE}/api/sessions/${encodeURIComponent(id)}`)
  if (!res.ok) throw new Error(`getSession: HTTP ${res.status}`)
  return (await res.json()) as SessionApi
}
```

### 3.6 `web/src/components/MainPage.tsx` — first real page

```tsx
import { useEffect, useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { createSession, listSessions } from '../lib/api'
import type { SessionApi } from '../lib/types'
import { STRINGS } from '../i18n/strings'

export default function MainPage() {
  const navigate = useNavigate()
  const [sessions, setSessions] = useState<SessionApi[] | null>(null)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    listSessions()
      .then(setSessions)
      .catch((e: Error) => setError(e.message))
  }, [])

  async function handleStart() {
    try {
      const { id } = await createSession()
      navigate(`/session/${id}`)
    } catch (e) {
      setError((e as Error).message)
    }
  }

  if (error) {
    return (
      <div className="rounded border border-red-300 bg-red-50 p-4" data-testid="error-banner">
        <p className="text-red-700">{STRINGS.errorPrefix}: {error}</p>
        <button
          onClick={() => window.location.reload()}
          className="mt-2 rounded bg-red-600 px-3 py-1 text-white"
        >
          {STRINGS.retry}
        </button>
      </div>
    )
  }

  if (sessions === null) {
    return <p data-testid="loading">{STRINGS.loading}</p>
  }

  return (
    <div>
      <button
        onClick={handleStart}
        data-testid="start-button"
        className="rounded bg-blue-600 px-4 py-2 text-white hover:bg-blue-700"
      >
        {STRINGS.startPractice}
      </button>

      {sessions.length === 0 ? (
        <p className="mt-8 text-slate-500" data-testid="empty-state">
          {STRINGS.emptyState}
        </p>
      ) : (
        <ul className="mt-8 space-y-3" data-testid="session-list">
          {sessions.map((s, i) => (
            <li
              key={s.id}
              data-testid="session-row"
              className="cursor-pointer rounded border bg-white p-4 shadow-sm hover:shadow"
              onClick={() => navigate(`/history/${s.id}`)}
            >
              <div className="flex items-baseline justify-between">
                <span className="font-mono text-sm text-slate-500">
                  #{sessions.length - i} · {new Date(s.startedAt).toLocaleString()}
                </span>
                {s.durationMin !== null && (
                  <span className="text-sm text-slate-500">
                    {s.durationMin} {STRINGS.minutesShort}
                  </span>
                )}
              </div>
              {s.summary && (
                <p className="mt-2 text-slate-700">
                  {s.summary.length > 80 ? `${s.summary.slice(0, 80)}…` : s.summary}
                </p>
              )}
              {s.keywords.length > 0 && (
                <div className="mt-2 flex flex-wrap gap-1">
                  {s.keywords.map((k) => (
                    <span
                      key={k}
                      className="rounded-full bg-slate-100 px-2 py-0.5 text-xs text-slate-600"
                    >
                      {k}
                    </span>
                  ))}
                </div>
              )}
            </li>
          ))}
        </ul>
      )}
    </div>
  )
}
```

## 4. CSS strategy

**Tailwind v4 zero-config** — `web/src/index.css` is one line:

```css
@import "tailwindcss";
```

The `@tailwindcss/vite` plugin generates utilities on-demand. No `tailwind.config.js`, no `postcss.config.js`. If we hit a class gap in v0.8.2, fall back to inline `style={{}}` (e.g., `style={{ minHeight: '80vh' }}`).

Dark mode toggle deferred to v0.8.5.

## 6. Dev / build / run workflow

### 6.1 First-time setup (one-time, documented in test report)

```bash
pnpm install                  # root deps (adds concurrently)
pnpm --dir web install        # web deps
npx playwright install chromium  # ~150MB download, cached afterwards
```

### 6.2 Development

```bash
pnpm dev-web
# → starts server (3000) + Vite (5173) concurrently
# Browser: http://localhost:5173 → full UI
# curl http://localhost:3000/api/sessions → raw JSON
```

### 6.3 Production build

```bash
pnpm build
# → 1. pnpm --dir web build  (vite → web/dist/)
# → 2. tsc                   (server → dist/)
```

### 6.4 E2E test

```bash
pnpm test:e2e
# → Playwright starts dev-web, waits for 5173, runs tests, kills
```

### 6.5 Preview production build

```bash
pnpm --dir web preview
# → serves web/dist/ on http://localhost:4173
# (Production SPA fallback in src/server.ts is v0.8.5; for now,
#  preview is the way to test the built bundle.)
```

## 8. Risk register (v0.8.2-specific)

| Risk | Mitigation |
|---|---|
| React 19 + Vite 6 peer-dep mismatch on Windows | Pin versions exactly; verify `pnpm --dir web install` succeeds first step |
| Tailwind v4 zero-config surprises | v4 `@tailwindcss/vite` plugin handles config; fallback to inline `style` for one-offs |
| `concurrently` quoting on Windows cmd.exe | Use `concurrently -k -n srv,web` with simple command strings; npm-run-all2 as fallback if needed |
| Playwright chromium download slow first time | One-time `npx playwright install chromium`; document in test report |
| E2E flake on dev-web startup | Playwright `webServer.timeout: 60_000` + `reuseExistingServer: !CI` |
| CORS during dev | Vite proxy `/api` → 3000; browser only sees 5173 |
| TypeScript strict + JSX + DOM lib conflict with server tsconfig | Separate `web/tsconfig.json`; root `tsconfig.json` excludes `web/` |
| Tooling fragmentation (web has its own deps + lockfile) | Acceptable per v0.8-design §2.2; user runs `pnpm --dir web install` once |
| Playwright `webServer` doesn't propagate child exit cleanly on Windows | `concurrently -k` propagates SIGTERM to both children; verified by v0.8.2 DoD |
